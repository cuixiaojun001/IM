86std::string Kernel::GetFileName(const char *path)
87{
88 int nlen = strlen(path);
89 if (nlen < 1) {
90 return std::string();
91 }
92 for (int i = nlen - 1; i >= 0; i --) {
93 if (path[i] == '\\' || path[i] == '/') {
94 return &path[i+1];
95 }
96 }
97 return std::string();

505void Kernel::slot_SendFile(int id, QString filename, uint64_t filesize)
506{
507 qDebug() << __func__ << " id: " << id << " filename:" << filename << " filesize:" << filesize;
508 STRU_FILE_INFO_RQ rq;
509 std::string strfilename = GetFileName(filename.toStdString().c_str());
510 strcpy_s(rq.szFileName, strfilename.c_str());
511 std::string strtime = QTime::currentTime().toString("hh_mm_ss_zzz").toStdString();
512 sprintf(rq.szFileId, "%s_%s", strfilename.c_str(), strtime.c_str());
513 rq.uuid = m_uuid;
514 rq.friendid = id;
515 rq.nFileSize = filesize;
516 // 将文件信息存储到FileInfo结构体中
517 FileInfo *info = new FileInfo;
518 info->nPos = 0;
519 info->nFileSize = filesize;
520 strcpy_s(info->szFileId, rq.szFileId);
521 strcpy_s(info->szFileName, strfilename.c_str());
522 strcpy_s(info->szFilePath, filename.toStdString().c_str());
523 if (fopen_s(&info->pFile, info->szFilePath, "rb") != 0) {
524 qDebug() << "主动发送文件端打开文件失败 errno=" << errno << " reason = " << strerror(errno);
525 return;
526 }
527 if (!info->pFile) {
